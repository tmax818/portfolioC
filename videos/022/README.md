## struct

```c

#include <stdio.h>
#include <string.h>

struct Student {
    char name[50];
    char id[10]
    int age;
    int grades[5];
};

int main(void){

    struct Student keven;

    kevin.age = 40;
    kevin.name = "Kevin"

    printf("Age: %d\n", keven.age);
    return 0;
}
```
